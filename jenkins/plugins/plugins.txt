# ==================================================
# Jenkins Plugins Configuration

# --------------------------------------------------
# Core Pipeline
# --------------------------------------------------
# workflow-aggregator → Enables Pipeline (Jenkinsfile support)
# pipeline-stage-view → Visual stage view in UI
workflow-aggregator
pipeline-stage-view

# --------------------------------------------------
# SCM (Source Control)
# --------------------------------------------------
# git → Allows Jenkins to clone repositories
git

# --------------------------------------------------
# Security / Credentials
# --------------------------------------------------
# credentials-binding → Secure usage of secrets in pipelines
credentials-binding

# --------------------------------------------------
# Test Reports
# --------------------------------------------------
# allure-jenkins-plugin → Integrates Allure test reports
# htmlpublisher → Publishes HTML reports in Jenkins (used for Allure report UI)
# email-ext → Advanced email notifications from pipelines (HTML body, dynamic subject, attachments)
# token-macro → Required by email-ext to render variables/macros in subject and body correctly
# mailer → Core SMTP mail sender used internally by email-ext to actually deliver emails
allure-jenkins-plugin
htmlpublisher
email-ext
token-macro
mailer

# --------------------------------------------------
# Infrastructure as Code / Job Management
# --------------------------------------------------
# configuration-as-code → Enables Jenkins configuration via YAML (JCasC), eliminating manual UI setup
# job-dsl → Enables creation of Jenkins jobs via code (Job DSL)
# structs → Required by job-dsl and complex parameter types (Active Choices) to survive Jenkins restarts
# uno-choice → Enables dynamic parameters (Active Choices) used for DEVICE combo
configuration-as-code
job-dsl
structs
uno-choice

# --------------------------------------------------
# UI / Visualization
# --------------------------------------------------
# blueocean → Provides a modern, user-friendly UI for Jenkins with enhanced pipeline visualization
blueocean

# --------------------------------------------------
# Logging
# --------------------------------------------------
# ansicolor → Enables ANSI color support in Jenkins console logs
# timestamper → Adds timestamps to each line of the Jenkins console output
ansicolor
timestamper

# --------------------------------------------------
# Workspace / Utilities
# --------------------------------------------------
# ws-cleanup → Allows cleaning the workspace in pipelines using cleanWs()
ws-cleanup
